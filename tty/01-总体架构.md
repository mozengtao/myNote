## 🏗️ 核心数据结构

### 内核态关键结构

```c
// 主要TTY结构 (include/linux/tty.h)
struct tty_struct {
    struct tty_driver *driver;     // TTY驱动
    struct tty_port *port;         // TTY端口
    struct tty_ldisc *disc;        // Line Discipline
    struct ktermios termios;       // 终端属性
    struct pid *session;           // Session ID
    struct pid *pgrp;              // Process Group ID
    // ...
};

// Line Discipline结构
struct tty_ldisc {
    struct tty_ldisc_ops *ops;     // 操作函数集
    struct tty_struct *tty;        // 关联的TTY
};

// 终端属性配置
struct ktermios {
    tcflag_t c_iflag;      // 输入模式标志
    tcflag_t c_oflag;      // 输出模式标志  
    tcflag_t c_cflag;      // 控制模式标志
    tcflag_t c_lflag;      // 本地模式标志
    cc_t c_cc[NCCS];       // 特殊字符数组
    speed_t c_ispeed;      // 输入波特率
    speed_t c_ospeed;      // 输出波特率
};

// 进程控制结构
struct signal_struct {
    pid_t pgrp;                    // 进程组ID
    pid_t session;                 // 会话ID
    struct tty_struct *tty;        // 控制终端
    // ...
};
```

### 用户态接口结构

```c
// termios.h - 用户态终端属性
struct termios {
    tcflag_t c_iflag;      // 输入标志
    tcflag_t c_oflag;      // 输出标志
    tcflag_t c_cflag;      // 控制标志
    tcflag_t c_lflag;      // 本地标志
    cc_t c_cc[NCCS];       // 控制字符
};

// 关键的控制字符索引
#define VINTR    0         // ^C (中断)
#define VQUIT    1         // ^\ (退出) 
#define VERASE   2         // 退格
#define VKILL    3         // 删除行
#define VEOF     4         // ^D (文件结束)
#define VSTART   8         // ^Q (开始)
#define VSTOP    9         // ^S (停止)
#define VSUSP    10        // ^Z (挂起)
```

## 🔄 数据流路径分析

### 输入数据流（Keyboard → Process）

```
1. 硬件中断
   键盘 → keyboard_interrupt() → 键码

2. TTY驱动层  
   键码 → tty_insert_flip_char() → TTY flip buffer

3. Line Discipline处理
   flip buffer → n_tty_receive_buf() → 字符处理
   - 特殊字符检测 (^C, ^Z, ^D)
   - Echo处理
   - 缓冲区管理

4. 用户空间读取
   read() → tty_read() → n_tty_read() → copy_to_user()
```

### 输出数据流（Process → Terminal）

```  
1. 系统调用
   write() → tty_write() → n_tty_write()

2. Line Discipline处理  
   字符 → 输出处理 (换行转换等)

3. 硬件输出
   → tty_driver.write() → 硬件显示
```

### 实验2：验证PTY机制

```c
// pty_test.c - 创建伪终端对
#include <stdlib.h>
#include <fcntl.h>
#include <unistd.h>
#include <stdio.h>

int main() {
    int master_fd;
    char *slave_name;
    
    // 打开master
    master_fd = posix_openpt(O_RDWR);
    if (master_fd == -1) {
        perror("posix_openpt");
        exit(1);
    }
    
    // 设置权限和解锁
    if (grantpt(master_fd) == -1) {
        perror("grantpt");
        exit(1);
    }
    
    if (unlockpt(master_fd) == -1) {
        perror("unlockpt");  
        exit(1);
    }
    
    // 获取slave设备名
    slave_name = ptsname(master_fd);
    if (slave_name == NULL) {
        perror("ptsname");
        exit(1);
    }
    
    printf("Master fd: %d\n", master_fd);
    printf("Slave device: %s\n", slave_name);
    
    // 测试数据传输
    write(master_fd, "Hello PTY\n", 10);
    
    return 0;
}
```

编译运行：
```bash
gcc -o pty_test pty_test.c
./pty_test
```

## 🚨 常见坑 & Debug方法

### 1. TTY vs PTY混淆
**问题**: 不理解物理终端和伪终端的区别
**Debug**: 
```bash
# 物理终端 (控制台)
ls -l /dev/tty[0-9]*
# 伪终端
ls -l /dev/pts/*

# 查看当前类型
tty | grep -E "(tty[0-9]|pts)"
```

### 2. Session/PGID混淆
**问题**: 不理解进程组和会话的关系
**Debug**:
```bash
# 查看完整进程关系树
ps -eo pid,ppid,pgid,sid,tty,cmd --forest

# 当前shell的session信息  
ps -p $$ -o pid,ppid,pgid,sid,tty,cmd
```

### 3. termios标志位理解错误
**问题**: 修改termios后行为异常
**Debug**:
```bash
# 保存当前设置
stty -g > /tmp/tty_backup

# 测试后恢复
stty $(cat /tmp/tty_backup)

# 或直接恢复默认
stty sane
```

### 4. 使用strace跟踪TTY操作
```bash
# 跟踪read/write系统调用
strace -e trace=read,write,ioctl -p $$

# 跟踪进程的TTY相关系统调用
strace -e trace=openat,ioctl,read,write,setsid,setpgid bash
```
